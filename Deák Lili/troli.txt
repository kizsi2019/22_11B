-- 16. feladat:
SELECT halozat.sorszam, megallok.nev AS "megallo"
FROM megallok, halozat, jaratok
WHERE megallok.id = halozat.megallo AND halozat.jarat = jaratok.id
	AND jaratok.jaratSzam = "City" AND halozat.irany = "A"
ORDER BY  1 ASC
